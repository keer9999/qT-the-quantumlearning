from qiskit import QuantumCircuit, QuantumRegister, ClassicalRegister

# Initialize circuit
qr = QuantumRegister(3, 'q')
cr = ClassicalRegister(3, 'c')
qc = QuantumCircuit(qr, cr)

# Add your gates here

# Measurement
qc.measure(qr, cr)
